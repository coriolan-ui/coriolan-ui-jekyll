Mixins

Clearfix

SCSS
        
.container {
    @include clearfix;
}
CSS Output
        
.container::after {
    clear: both;
    content: "";
    display: table;
}

Ellipsis

SCSS
        
.title {
    @include ellipsis;
}
CSS Output
        
.title {
    white-space: nowrap;
    text-overflow: ellipsis;
    overflow: hidden;
    display: block;
}

Placeholder

SCSS
        
textarea {
    width: 100%;
    font-size: 16px;

    @include placeholder {
        font-size: 12px;
    }
}
CSS Output
        
textarea {
    width: 100%;
    font-size: 16px;
}

textarea::-webkit-input-placeholder {
    font-size: 12px;
}

textarea::-moz-placeholder {
    font-size: 12px;
}

textarea:-moz-placeholder {
    font-size: 12px;
}

textarea:-ms-input-placeholder {
    font-size: 12px;
}
